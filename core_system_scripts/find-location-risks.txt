Location risk
---------------
SELECT * FROM UW_t_POLICIES WHERE POL_SEQ_NO IN (

SELECT  plc_pol_seq_no
 FROM UW_t_POL_LOCATIONS
WHERE plc_loc_code IN (SELECT DISTINCT loc_code FROM uw_r_locations
where 

--loc_code ='000DM0022807'
loc_earthqk='MMR017024'

))
AND 

pol_cla_code ='FI'

select pol_policy_no, cus_code, cus_indv_surname, 
adr_city,adr_street, 
adr_building, 
adr_number, adr_province, adr_district, adr_postal_code, adr_earthqk, adr_cyclone
